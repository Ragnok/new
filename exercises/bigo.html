Estimate the time complexity of the following code. Give best, worst, and average case:

void bigo1(int *arr, size_t len)
{
	int z = 0;
	size_t pos = len;
	while(pos--) {
		if(arr[pos] < 0) {
			arr[pos] = 0;
			pos = len;
		}
	}
}

Estimate the time complexity of the following code. Give best, worst, and average case:

void bigos(char *str)
{
	size_t len = strlen(str);
	while(len) {
		printf("%c\n", str[len-1]);
		len /= 4;
	}
}

Estimate the time complexity of the following code. Give best, worst, and average case:

int bigo4(int limit)
{
	int total = 0;
	for(int a = 1; a < limit; a *= 2) {
		for(int b = limit; b > 0; b /= 2) {
			for(int c = b; c < limit; c += 2) {
				total += a + b*c;
			}
		}
	}
	return total;
}
